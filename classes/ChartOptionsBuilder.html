import { ChartOptions, LegendItem } from "chart.js/auto";
import { ForecastContext } from "../interfaces/forecast-interface";
import { I18nService } from "../../shared/services/i18n.service";

export class ChartOptionsBuilder {

  constructor(
    private i18nService: I18nService
  ) { }

  build(ctx: ForecastContext): ChartOptions<'line'> {
    return {
      responsive: true,
      maintainAspectRatio: false,
      layout: {
        padding: { left: 5, right: 30, top: 15, bottom: 0 }
      },
      plugins: {
        legend: {
          display: true,
          position: 'bottom',
          labels: {
            filter: (item: LegendItem) =>
              item.datasetIndex !== undefined && item.datasetIndex < 4
          }
        },
        tooltip: {
          enabled: true,
          mode: 'index',
          callbacks: {
            label: (ctx) => {
              const label = ctx.dataset.label ? ctx.dataset.label + ': ' : '';
              return `${label} R$ ${(Number(ctx.raw)).toLocaleString(this.i18nService.currentLang)}`;
            },
            title: items => items[0].label
          }
        }
      },
      hover: { mode: 'nearest', intersect: true },
      scales: {
        y: {
          ticks: {
            callback: (value: string | number) => this.formatMoney(Number(value))
          },
          grid: { color: '#cccccc55' },
          border: { color: '#cccccc55' }
        },
        x: { grid: { display: false } }
      }
    };
  }

  formatMoney(value: number): string {
    const { valueRound, words } = this.transformInWords(value);

    return `R$ ${valueRound}${words ? ' ' + words : ''}`;
  }

  private scaleMap = [
    { limit: 1_000_000_000_000, divisor: 1_000_000_000_000, singularKey: 'trillion', pluralKey: 'trillions' },
    { limit: 1_000_000_000, divisor: 1_000_000_000, singularKey: 'billion', pluralKey: 'billions' },
    { limit: 1_000_000, divisor: 1_000_000, singularKey: 'million', pluralKey: 'millions' },
    { limit: 1_000, divisor: 1_000, singularKey: 'thousand', pluralKey: 'thousand' }
  ];

  transformInWords(money: number): { valueRound: string; words: string } {
    const absValue = Math.abs(money);

    for (const scale of this.scaleMap) {
      if (absValue >= scale.limit) {
        const scaled = absValue / scale.divisor;

        const literalKey = scaled > 1 ? scale.pluralKey : scale.singularKey;
        const words = literalKey ? this.i18nService.t(literalKey) : '';

        // Formato con 1 decimal para evaluar si termina en .0
        const raw = scaled.toLocaleString(
          this.i18nService.currentLang,
          { minimumFractionDigits: 1, maximumFractionDigits: 1 }
        );

        // Detectar si el número termina en ",0" o ".0" según locale
        const sep = raw.includes(',') ? ',' : '.';
        const cleaned = raw.endsWith(`${sep}0`)
          ? raw.slice(0, -2)  // remover ",0" o ".0"
          : raw;

        return {
          valueRound: `${money < 0 ? '-' : ''}${cleaned}`,
          words
        };
      }
    }

    // Sin escala (menor a mil)
    return {
      valueRound: `${money < 0 ? '-' : ''}${absValue.toLocaleString(
        this.i18nService.currentLang,
        { minimumFractionDigits: 0, maximumFractionDigits: 0 }
      )}`,
      words: ''
    };
  }
}
