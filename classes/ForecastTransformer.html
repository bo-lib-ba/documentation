import { ChartDataset } from "chart.js/auto";
import { ForecastContext, ForecastObject } from "../interfaces/forecast-interface";
import { I18nService } from "../../shared/services/i18n.service";

export class ForecastTransformer {

  constructor(private i18nService: I18nService){}

  toContext(context: ForecastObject): ForecastContext {

    const labelsHistory = Object.keys(context.current_revenues);
    const labelsForecast = context.revenue.line.map((l: { competence: any; }) => l.competence);
    const dataLabels = [...labelsHistory, ...labelsForecast];

    // Identify the last historical month
    const lastHistoryLabel = labelsHistory[labelsHistory.length - 1];

    const lastRevenueHistory = context.current_revenues[lastHistoryLabel] ?? null;
    const lastExpenseHistory = context.current_expenses[lastHistoryLabel] ?? null;

    // Build forecast arrays WITH bridge included
    const revenueForecastData = dataLabels.map(l => {
      const item = context.revenue.line.find((r: { competence: any; }) => r.competence === l);
      return item ? item.value : null;
    });

    const expenseForecastData = dataLabels.map(l => {
      const item = context.expense.line.find((r: { competence: any; }) => r.competence === l);
      return item ? item.value : null;
    });

    // Inject bridge point: replace the PREVIOUS null before first forecast with last history value
    const firstForecastIndex = revenueForecastData.findIndex(v => v !== null);
    if (firstForecastIndex > 0) {
      revenueForecastData[firstForecastIndex - 1] = lastRevenueHistory;
      expenseForecastData[firstForecastIndex - 1] = lastExpenseHistory;
    }

    const datasets: ChartDataset<'line'>[] = [
      {
        label: this.i18nService.t('revenues'),
        data: dataLabels.map(l => context.current_revenues[l] ?? null),
        borderColor: '#00B28E',
        fill: false,
        tension: 0
      },
      {
        label: this.i18nService.t('expenses'),
        data: dataLabels.map(l => context.current_expenses[l] ?? null),
        borderColor: '#C64840',
        fill: false,
        tension: 0
      },
      {
        label: this.i18nService.t('revenueForecast'),
        data: revenueForecastData,
        borderColor: '#00B28E',
        borderDash: [10, 5],
        fill: false,
        spanGaps: true,
        tension: 0
      },
      {
        label: this.i18nService.t('expenseForecast'),
        data: expenseForecastData,
        borderColor: '#C64840',
        borderDash: [10, 5],
        fill: false,
        spanGaps: true,
        tension: 0
      },
      {
        label: this.i18nService.t('revenueMax'),
        data: dataLabels.map(l => {
          const item = context.revenue.interval.find((i: { competence: any; }) => i.competence === l);
          return item ? item.high_interval : null;
        }),
        borderColor: 'rgba(2, 117, 60, 0.4)',
        fill: false,
        tension: 0
      },
      {
        label: this.i18nService.t('revenueMin'),
        data: dataLabels.map(l => {
          const item = context.revenue.interval.find((i: { competence: any; }) => i.competence === l);
          return item ? item.low_interval : null;
        }),
        borderColor: 'rgba(2, 117, 60, 0.4)',
        backgroundColor: 'rgba(181, 181, 181, 0.2)',
        fill: 4,
        tension: 0
      },
      {
        label: this.i18nService.t('expenseMax'),
        data: dataLabels.map(l => {
          const item = context.expense.interval.find((i: { competence: any; }) => i.competence === l);
          return item ? item.high_interval : null;
        }),
        borderColor: 'rgba(117, 2, 2, 0.4)',
        fill: false,
        tension: 0
      },
      {
        label: this.i18nService.t('expenseMin'),
        data: dataLabels.map(l => {
          const item = context.expense.interval.find((i: { competence: any; }) => i.competence === l);
          return item ? item.low_interval : null;
        }),
        borderColor: 'rgba(117, 2, 2, 0.4)',
        backgroundColor: 'rgba(181, 181, 181, 0.2)',
        fill: 6,
        tension: 0
      }
    ];

    const labels: any[] = ForecastTransformer.formatLabels(
      dataLabels,
      context.periodType, // 'monthly' | 'weekly'
      this.i18nService
    );

    return {
      labels,
      datasets,
      revenueAccuracyCod: context.revenue.mapecategory,
      expenseAccuracyCod: context.expense.mapecategory,
      insightDate: context.mdmLastUpdated,
      errorMessage: context.messageWarning || '',
      loading: true,
      hasError: false
    };
  }

  static formatMonthLabel(dateStr: string, i18n: I18nService): string {
    const date = new Date(`${dateStr.length === 7 ? dateStr + '-01' : dateStr}T00:00:00`);

    // Obtenemos el nombre traducido del mes vía PO-i18n
    const monthKey = date.toLocaleString('en-US', { month: 'short' }).toLowerCase();
    const translatedMonth = i18n.t(monthKey);

    const year = date.getFullYear();

    return `${translatedMonth} ${year}`;
  }

  static formatWeekLabel(range: string): string {
    const [start, end] = range.split('|');

    const formatter = new Intl.DateTimeFormat('pt-BR', {
      day: '2-digit',
      month: '2-digit',
      year: '2-digit'
    });

    const startLabel = formatter.format(new Date(`${start}T00:00:00`));
    const endLabel = formatter.format(new Date(`${end}T00:00:00`));

    return `${startLabel} - ${endLabel}`;
  }

  static formatLabels(
    keys: string[],
    mode: 'monthly' | 'weekly',
    i18n: I18nService
  ): string[] {

    if (mode === 'monthly') {
      return keys
        .sort((a, b) => new Date(a).getTime() - new Date(b).getTime())
        .map(k => this.formatMonthLabel(k, i18n));
    }

    if (mode === 'weekly') {
      return keys
        .sort((a, b) => {
          const aStart = new Date(a.split('|')[0]);
          const bStart = new Date(b.split('|')[0]);
          return aStart.getTime() - bStart.getTime();
        })
        .map(k => this.formatWeekLabel(k));
    }

    return keys;
  }
