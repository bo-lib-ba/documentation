export class ChatComponent {
  /** Elemento de scroll da área de mensagens */
  @ViewChild('scrollContainer') scrollContainer!: ElementRef;

  /** Elemento raiz da janela do chat, usado para resetar transformações */
  @ViewChild('chatWrapper') chatWrapper!: ElementRef<HTMLDivElement>;
  @ViewChild(CdkDrag) cdkDrag!: CdkDrag;

  @Input() incomingMessage: string | null = null;
  @Input() darkMode?: boolean | null = false;
  @Input() sunsetMode?: boolean | null = false;
  @Input() idRequest: string | null = null;
  @Input() quickReplies: string[] | null = null;
  @Input() linksToOpen: string[] | null = null;

  @Output() userMessage = new EventEmitter<string>();
  @Output() openLink = new EventEmitter<string>();

  showGif = true;

  /** Define se o chat está visível (true) ou minimizado (false) */
  isOpen = false; // controla o estado animado (aberto ou fechado)
  chatVisible = false; // controla se o conteúdo interno está visível
  thinking: boolean = false; // controla o estado de "pensando" do bot
  /** Define se o chat está em modo tela cheia (maximizado) */
  isMaximized = false;

  /** Texto digitado ou preenchido automaticamente pelo usuário */
  inputMessage = '';

  /** Lista de mensagens trocadas entre usuário e bot */
  // messages: { from: 'user' | 'bot'; text: string }[] = [];
  messages: ChatMessage[] = [];

  lastLinks: string[] = [];

  /** Lista de perguntas rápidas sugeridas pelo bot */
  quickRepliesOptions: string[] = [];

  sunsetPallete = [
    { value: '#a5131a' },
    { value: '#b23a40' },
    { value: '#bf6166' },
    { value: '#cc888c' },
    { value: '#d9afb2' },
    { value: '#e6d6d8' },
    { value: '#f0e1e2' },
    { value: '#f5ebec' },
    { value: '#faf5f5' },
    { value: '#ffffff' }
  ];

  defaultPallete = [
    { value: '#045b8f' },
    { value: '#2e76a3' },
    { value: '#5891b7' },
    { value: '#82acca' },
    { value: '#acc7de' },
    { value: '#d6e2f1' },
    { value: '#e3ecf6' },
    { value: '#eff5fa' },
    { value: '#f7fafc' },
    { value: '#ffffff' }
  ];

  PoChartType = PoChartType;
  notificationAudio = new Audio('assets/notification.mp3'); // ajuste o caminho conforme necessário


  ngOnChanges(changes: SimpleChanges): void {
    if (this.checkUpdatesMessages(changes)) {
      const raw = changes['incomingMessage'].currentValue as string;

      let parsed: any;
      let chatMsg: ChatMessage;

      try {
        parsed = JSON.parse(raw);
        chatMsg = this.processAgentMessage(parsed);

      } catch {
        chatMsg = {
          from: 'bot',
          type: 'text',
          text: raw
        };
      }

      this.messages.push(chatMsg);

      this.playNotificationSound();
      this.scrollToBottom();
      this.thinking = false;
      this.lastLinks = this.linksToOpen ?? [];
    }
    if(this.checkQuickReplies(changes)) {
      this.quickRepliesOptions = changes['quickReplies'].currentValue as string[];
      console.log(this.quickRepliesOptions);
      this.scrollToBottom();
      this.thinking = false;
    }
  }

  checkUpdatesMessages(changes: SimpleChanges): boolean {
    return (changes['incomingMessage'] && changes['incomingMessage'].currentValue) || (changes['idRequest'] && changes['idRequest'].currentValue);
  }

  checkQuickReplies(changes: SimpleChanges): boolean {
    return (changes['quickReplies'] && changes['quickReplies'].currentValue) || (changes['idRequest'] && changes['idRequest'].currentValue);
  }

  // Método para tocar o som
  playNotificationSound(): void {
    if (document.hidden) { // Só toca se a aba NÃO estiver em foco
      this.notificationAudio.currentTime = 0;
      this.notificationAudio.play().catch(() => {});
    }
  }

  /**
   * Envia uma pergunta rápida como se fosse uma mensagem digitada pelo usuário
   * @param text Texto da pergunta sugerida
   */
  sendQuickMessage(text: string): void {
    this.inputMessage = text;
    this.sendMessage();
    this.quickRepliesOptions = []; // Limpa as opções de perguntas rápidas
  }

  openChat() {
    this.isOpen = true;
    setTimeout(() => (this.chatVisible = true), 400); // espera animação terminar
  }

  closeChat() {
    this.chatVisible = false;
    setTimeout(() => (this.isOpen = false), 250); // espera animação reversa
  }

  /**
   * Envia a mensagem digitada pelo usuário e simula uma resposta do bot
   */
  sendMessage(): void {
    this.thinking = true;

    const trimmed = this.inputMessage.trim();
    if (!trimmed) return;

    // Mensagem do usuário
    this.messages.push({ from: 'user', text: trimmed, type: 'text' });
    this.userMessage.emit(trimmed); // <-- avisa o pai

    this.inputMessage = '';
    this.scrollToBottom();
  }

  openLinkExternal(link: string): void {
    this.openLink.emit(link); // <-- avisa o pai para abrir o link
  }

  /**
   * Alterna entre janela normal e maximizada
   * Remove transformações de drag antes de maximizar
   */
  toggleMaximize(): void {
    if (!this.isMaximized && this.chatWrapper) {
      // Remove transform inline
      this.chatWrapper.nativeElement.style.transform = 'none';
      // Reseta a posição do drag
      if (this.cdkDrag) {
        this.cdkDrag.reset();
      }
    }
    this.isMaximized = !this.isMaximized;
  }

  /**
   * Rola a área de mensagens para o final (última mensagem)
   */
  scrollToBottom(): void {
    setTimeout(() => {
      if(this.scrollContainer){
        const el = this.scrollContainer.nativeElement;
        el.scrollTop = el.scrollHeight;
      }
    });
  }

  processAgentMessage(agentJson: any): ChatMessage {
    const type = agentJson.type;

    switch (type) {
      case 'text':
        return {
          from: 'bot',
          type: 'text',
          text: agentJson.text || ''
        };

      case 'table':
        return {
          from: 'bot',
          type: 'table',
          text: agentJson.text,
          tableColumns: agentJson.table?.columns ?? [],
          tableData: agentJson.table?.items ?? []
        };

      case 'bo-drilldown':
        return {
          from: 'bot',
          type: 'bo-drilldown',
          text: agentJson.text,
          drillDownData: {
            header: agentJson.drillDownData.header,
            items: agentJson.drillDownData.items,
            hasNext: agentJson.drillDownData.hasNext
          }
        };

      case 'chart':
        return {
          from: 'bot',
          type: 'chart',
          text: agentJson.text,
          chartType: agentJson.chart?.type, // pie, line etc
          chartData: agentJson.chart.data.map( (res:any,index:number) => ({
            label: res.label,
            data: res.value,
            color: this.sunsetMode ? this.sunsetPallete[index].value : this.defaultPallete[index].value
          }))
        };

      default:
        return {
          from: 'bot',
          type: 'text',
          text: 'Tipo de mensagem não reconhecido.'
        };
    }
  }

  getDynamicHeight(items: any[]): number {
    if (!items){
      return 200;
    }
    const rows = items.length;
    return Math.min(800, 200 + rows * 35);
}

}
