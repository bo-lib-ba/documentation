import { CommonModule } from '@angular/common';
import { Component, ElementRef, EventEmitter, Output, ViewChild } from '@angular/core';
import { PoModalComponent, PoModule, PoSwitchLabelPosition } from '@po-ui/ng-components';
import { Chart, ChartOptions } from 'chart.js/auto';
import { ForecastTransformer } from './util/forecast-transformer';
import { ChartOptionsBuilder } from './util/chart-options.builder';
import { ForecastContext, ForecastObject } from './interfaces/forecast-interface';
import { I18nService } from '../shared/services/i18n.service';
import { I18nPipe } from "../shared/pipes/i18n.pipe";
import { format } from 'date-fns';

@Component({
  selector: 'bo-forecast-modal',
  templateUrl: './forecast-modal.component.html',
  styleUrls: ['./forecast-modal.component.css'],
  standalone: true,
  imports: [
    CommonModule,
    PoModule,
    I18nPipe
  ]
})
export class ForecastModalComponent {

  @ViewChild('forecastModal') forecastModal!: PoModalComponent;
  @ViewChild('forecastChart') forecastChart!: ElementRef<HTMLCanvasElement>;

  @Output() toggleChange = new EventEmitter<'visible' | 'hidden'>();
  @Output() onClose = new EventEmitter<'closed'>();

  forecastContext: ForecastContext = {
    labels: [],
    datasets: [],
    revenueAccuracyCod: '',
    expenseAccuracyCod: '',
    insightDate: '',
    errorMessage: '',
    loading: false,
    hasError: false,
  };
  
  forecastObject: ForecastObject = {
    current_expenses: {},
    current_revenues: {},
    expense: {
      accuracy: 0,
      mapecategory: '',
      interval: [],
      line: []
    },
    revenue: {
      accuracy: 0,
      mapecategory: '',
      interval: [],
      line: []
    },
    mdmLastUpdated: '',
    Code: '',
    branches: '',
    periodType: 'monthly'
  };

  chart!: Chart;
  showInterval = false;
  switchIntervalLabelPosition = PoSwitchLabelPosition.Left;
  insightDate: string = '';

  constructor(
    private i18nService: I18nService
  ) {}

  open(context: ForecastObject): void {
    this.forecastObject = context;
    const transformer = new ForecastTransformer(this.i18nService);
    this.forecastContext = transformer.toContext(context);
    
    if (context.mdmLastUpdated) {
      this.insightDate = format(
        new Date(context.mdmLastUpdated),
        'dd/LL/yyyy hh:mm:ss'
      );
    }

    this.forecastModal.open();
    setTimeout(() => this.buildChart(), 200);
  }

  close(): void {
    if (this.chart) this.chart.destroy();
    this.onClose.emit('closed');
    this.forecastModal.close();
    this.showInterval = false;
  }

  private buildChart(): void {
    if (!this.forecastContext || !this.forecastChart) return;

    const ctx = this.forecastChart.nativeElement.getContext('2d');
    if (!ctx) return;

    const builder = new ChartOptionsBuilder(this.i18nService);
    const options: ChartOptions<'line'> = builder.build(this.forecastContext);

    if (this.chart) this.chart.destroy();
    this.chart = new Chart(ctx, {
      type: 'line',
      data: {
        labels: this.forecastContext.labels,
        datasets: this.forecastContext.datasets
      },
      options
    });

    setTimeout(() => {
      this.toggleInterval(true);
      this.forecastContext.loading = false;
    }, 200);
  }

  toggleInterval(prevent?: boolean): void {
    if (!this.chart) {
      return;
    }

    if(!prevent){
      this.toggleChange.emit(this.showInterval ? 'visible' : 'hidden');
    }

    const intervalIndexes = [4, 5, 6, 7];

    intervalIndexes.forEach(i => {
      this.chart.data.datasets[i].hidden = !this.showInterval;
    });

    if (this.chart.options.scales && this.chart.options.scales['y']) {
      this.chart.options.scales['y'].min = 0;
    }

    this.chart.update('active');
  }

  formatAccuracyText(code: string): string {
    const map: Record<string, string> = {
      '01': this.i18nService.t('excellent'), 
      '02': this.i18nService.t('good'), 
      '03': this.i18nService.t('fair'),
      '04': this.i18nService.t('low'), 
    };
    return map[code] ?? 'N/A';
  }
}
